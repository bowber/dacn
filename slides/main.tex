% SLIDE 8: Calculation Results
% ============================================================
\begin{frame}{Kết quả tính toán}
    \begin{tikzpicture}
        \node[rectangle, fill=darkblue, inner sep=12pt, text width=0.92\textwidth, minimum height=6cm, anchor=north west] at (4pt, -4pt) {};
        \node[rectangle, fill=white, draw=darkblue, line width=2pt, inner sep=12pt, text width=0.92\textwidth, minimum height=6cm, anchor=north west] at (0,0) {
            \begin{columns}[T]
                \begin{column}{0.48\textwidth}
                    \textbf{\textcolor{bkblue}{Cân bằng năng lượng:}}
                    
                    \vspace{0.3em}
                    $Q_{\text{hóa hơi}} = \dot{m} \cdot C_p \cdot \Delta T$
                    
                    \vspace{0.5em}
                    \textbf{\textcolor{bkblue}{Lưu lượng nước tối ưu:}}
                    
                    \vspace{0.3em}
                    $\dot{m} = \dfrac{Q_{\text{hóa hơi}}}{C_p \cdot (T_{out} - T_{in})}$
                    
                    \vspace{0.5em}
                    $= \dfrac{5282}{4186 \times (68 - 30)}$
                    
                    \vspace{0.5em}
                    $= 0.033$ kg/s = \textbf{2.0 L/min}
                \end{column}
                \begin{column}{0.48\textwidth}
                    \textbf{\textcolor{bkblue}{Thông số đầu vào:}}
                    \begin{itemize}
                        \item $Q_{\text{hóa hơi}} = 5282$ W
                        \item $C_p = 4186$ J/(kg·K)
                        \item $T_{in} = 30$°C
                        \item $T_{out} = 68$°C
                    \end{itemize}
                    
                    \vspace{0.3em}
                    \textbf{\textcolor{bkblue}{Kết quả:}}
                    \begin{itemize}
                        \item Độ mở van: \textbf{28\%}
                        \item Tiết kiệm: \highlight{72\%} lưu lượng
                    \end{itemize}
                \end{column}
            \end{columns}
        };
    \end{tikzpicture}
\end{frame}
